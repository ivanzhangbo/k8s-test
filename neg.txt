apiVersion: v1
kind: ServiceAccount
metadata:
  name: my-ksa
# Secretを作成しておき、ここで指定すると、
# このSAを使うすべてのPodで自動的にこの認証情報が使われます
imagePullSecrets:
  - name: my-docker-registry-secret

---
apiVersion: v1
kind: Service
metadata:
  name: backend-service # ★重要: この名前がDNS名になります
spec:
  type: ClusterIP       # ★ここがNodePortではなくClusterIPになります（省略可）
  selector:
    app: backend-app    # このラベルを持つPodに転送する
  ports:
    - protocol: TCP
      port: 80          # Serviceが待ち受けるポート (アクセス元はここを指定)
      targetPort: 8080  # 実際のコンテナが動いているポート
spec:
      # ★ここで作成した ServiceAccount を指定します
      serviceAccountName: my-custom-sa
