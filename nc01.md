# ✅ **完成スクリプト（完全動作版）**

以下を `maintenance.sh` として保存してください：

```bash
#!/bin/bash

CONF_FILE="/etc/httpd/conf/httpd.conf"

if [ $# -ne 1 ]; then
    echo "Usage: $0 {1|2}"
    exit 1
fi

MODE=$1

case "$MODE" in
1)
    echo "=== メンテナンスモード ON ==="

    # ProxyPass をコメントアウト
    sed -i 's|^\s*ProxyPass |#ProxyPass |' "$CONF_FILE"
    sed -i 's|^\s*ProxyPassReverse |#ProxyPassReverse |' "$CONF_FILE"

    # ErrorDocument のコメント解除（有効化）
    sed -i 's|^\s*#ErrorDocument 404 /maintenance/maintenance.html|ErrorDocument 404 /maintenance/maintenance.html|' "$CONF_FILE"
    ;;

2)
    echo "=== メンテナンスモード OFF ==="

    # ProxyPass のコメント解除
    sed -i 's|^\s*#ProxyPass |ProxyPass |' "$CONF_FILE"
    sed -i 's|^\s*#ProxyPassReverse |ProxyPassReverse |' "$CONF_FILE"

    # ErrorDocument をコメントアウト（無効化）
    sed -i 's|^\s*ErrorDocument 404 /maintenance/maintenance.html|#ErrorDocument 404 /maintenance/maintenance.html|' "$CONF_FILE"
    ;;

*)
    echo "Usage: $0 {1|2}"
    exit 1
    ;;
esac

# Apache reload
systemctl reload httpd
echo "Apache reloaded."
```
